CREATE DATABASE IF NOT EXISTS exam_db;
USE exam_db;

-- 2. Таблица пользователей (расширенная)
CREATE TABLE IF NOT EXISTS users (
    id INT AUTO_INCREMENT PRIMARY KEY,
    login VARCHAR(50) NOT NULL UNIQUE,
    email VARCHAR(100) NOT NULL UNIQUE,
    password VARCHAR(255) NOT NULL,
    full_name VARCHAR(100) NOT NULL,
    phone VARCHAR(20) NOT NULL,
    birth_date DATE,
    is_admin INT DEFAULT 0,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

-- 3. Таблица заявок (универсальная)
CREATE TABLE IF NOT EXISTS applications (
    id INT AUTO_INCREMENT PRIMARY KEY,
    user_id INT NOT NULL,
    service_type VARCHAR(100) NOT NULL,  -- 'банкетный зал' или 'катер/яхта/лайнер'
    event_date DATE NOT NULL,
    payment_method VARCHAR(50) NOT NULL,
    status VARCHAR(50) DEFAULT 'Новая',
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (user_id) REFERENCES users(id) ON DELETE CASCADE
);

-- 4. Таблица отзывов
CREATE TABLE IF NOT EXISTS reviews (
    id INT AUTO_INCREMENT PRIMARY KEY,
    user_id INT NOT NULL,
    application_id INT NOT NULL,
    review TEXT NOT NULL,
    rating INT DEFAULT 5,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (user_id) REFERENCES users(id) ON DELETE CASCADE,
    FOREIGN KEY (application_id) REFERENCES applications(id) ON DELETE CASCADE
);

1. Активируй нужную библиотеку (самое важное!)
Слева внизу нажми "More Shapes" (или «Ещё фигуры»)

В разделе "Software" поставь галочку напротив "Entity Relation" 

Нажми "Apply"

Теперь у тебя появятся готовые фигуры для ER-диаграмм: сущности (таблицы), атрибуты (поля), ключи, связи.

2. Начни с шаблона (проще всего)
В меню выбери Arrange → Insert → Template 

В поиске введи "Entity Relationship" или найди в разделах:

Basic → есть простой шаблон с тремя таблицами

Software → более сложные примеры ER-диаграмм 

Шаблон можно вставить в текущую диаграмму и сразу начать редактировать под свой проект.

3. Построй свою диаграмму
Перетащи на холст сущности (таблицы) из левой панели

Используй готовую фигуру "Table" для каждой таблицы (users, applications, reviews)

Добавь поля (id, login, user_id, status и т.д.)

Для связей используй линии с обозначениями: 1 --- ∞ (один ко многим)

4. Быстрое редактирование таблиц (удобная фишка!)
Выдели таблицу → справа в панели "Arrange" найди кнопки:

Insert Row After/Before — добавить строку (поле) в таблицу 

Или просто нажми Ctrl+Enter (Windows), чтобы скопировать строку 

5. Сохрани результат
File → Export as → выбери PNG или PDF

Готовую картинку можно вставить в отчёт или сохранить на флешку
